\subsection{Quadratic Formular}
\begin{gather}
    ax^2+bx+c = a \bracket[s]{x^2 + \frac{b}{a}x} + c
    = a \bracket[s]{x^2 + \frac{b}{a}x + \bracket{\frac{b}{2a}}^2 - \bracket{\frac{b}{2a}}^2} + c\\
    = a \bracket[s]{\bracket{x + \frac{b}{2a}}^2 - \frac{b^2}{4a^2}} +c 
    = a \bracket[]{x + \frac{b}{2a}}^2 - \frac{b^2}{4a} +c\\
    ax^2 + bx + c = 0 \\
    a \bracket[]{x + \frac{b}{2a}}^2 - \frac{b^2}{4a} +c = 0\\
    a \bracket[]{x+\frac{b}{2a}}^2 = \frac{b^2}{4a} -c \\
    x + \frac{b}{2a} = \pm \sqrt{\frac{b^2}{4a^2}-\frac{c}{a}}\\
    x = -\frac{b}{2a} \pm \sqrt{\frac{b^2}{4a^2}-\frac{4ac}{4a^2}} 
    = -\frac{b}{2a} \pm \frac{1}{2a} \sqrt{b^2-4ac}\\
    \result{x=\frac{-b \pm \sqrt{b^2-4ac}}{2a} }
\end{gather}
