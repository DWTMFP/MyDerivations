\section{Sums}

\subsection{Sum of the first $n$ even numbers}
Note: not my idea, but my way of showing this relation\\
Let $m$ be the $n$th even number, then $m=2n,\quad m,n \in \N$ 
\begin{gather}
\sum^n_{k = 1}(2k) = 2 \cdot \sum^n_{k = 1}(k) = 2 \cdot \frac{n(n + 1)}{2} = n^2 + n\\
\result{\sum^n_{k = 1}(2k) = n^2 + n} \label{eq:res:sum_of_first_n_even_numbers}
\end{gather}

\subsection{Sum of the first $n$ odd numbers}
Note: not my idea, but my way of showing this relation\\
Let $m$ be the $n$th odd number, then $m=2n-1,\quad m,n \in \N$\\
Using Equation \ref{eq:res:sum_of_first_n_even_numbers}
\begin{gather}
\sum^n_{k = 1}(2k - 1) = \sum^n_{k = 1}(2k) - \sum^n_{k = 1}(1) = n^2 + n - n\\
\result{\sum^n_{k = 1}(2k - 1) = n^2} \label{eq:res:sum_of_first_n_odd_numbers}
\end{gather}

\subsection{Sum of the first even numbers up to $n$}
If we let an even number $n\in\N$ have the form $n=2k, k\in\N_0$ we need $2k=n$ for the upper bound of the sum,
which implies $k=\nicefrac{n}{2}$.
\begin{gather}
0 + 2 + 4 +\dots + n = \sum_{k = 1}^{\nicefrac{n}{2}}(2k) \\
\mbox{let } m = \frac{n}{2} \\
\implies \sum_{k = 1}^{\nicefrac{n}{2}}(2k) = \sum_{k = 1}^{m}(2k)\\
\mbox{From equation \ref{eq:res:sum_of_first_n_even_numbers} we get}\\
\sum_{k = 1}^{m}(2k) = m^2 + m = \frac{n^2}{4} + \frac{n}{2} = \frac{n^2 + 2n}{4} \\
\result{\sum_{k = 1}^{\nicefrac{n}{2}}(2k) = 0 + 2 + 4 +\dots + n = \frac{n(n + 2)}{4}} 
\end{gather}

\subsection{Sum of the first odd numbers up to $n$}
If we let an odd number $n\in\N$ have the form $n=2k-1, k\in\N_0$ we need $2k-1=n$ for the upper bound of the sum,
which implies $k=\nicefrac{(n+1)}{2}$.
\begin{gather}
1 + 3 +\dots{} + n =\sum_{k = 1}^{\nicefrac{\bracket{n + 1}}{2}}(2k - 1) \\
\mbox{let } m = \frac{n + 1}{2}, m\in\N\\
\implies \sum_{k = 1}^{\nicefrac{\bracket{n + 1}}{2}}(2k - 1) = \sum_{k = 1}^{m}(2k - 1)\\
\mbox{From equation \ref{eq:res:sum_of_first_n_odd_numbers} we get}\\
\sum_{k = 1}^{m}(2k - 1) = m^2 = \bracket{\frac{n + 1}{2}}^2 \\
\result{\sum_{k = 1}^{\nicefrac{\bracket{n + 1}}{2}}(2k - 1) = 1 + 3 +\dots{} + n = \bracket{\frac{n + 1}{2}}^2}
\end{gather}
