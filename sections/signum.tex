\section{Signum Function}
The signum function yields the sign of a number:
\[
\sgnfn{x} = 
\begin{cases} 
    +1 &\mbox{if }x>0 \\
    \hfil0 &\mbox{if }x=0 \\
    -1 &\mbox{if }x<0
\end{cases} \label{eq:sgn_def}
\]

\subsection{link to $\abs{x}$}
\begin{gather}
    \begin{aligned}
    \abs{x}' &= \bracket{\sqrt{x^2}}' = \frac{1}{\cancel{2} \sqrt{x^2}} \cdot \bracket{\cancel{2} x} = \frac{x}{\sqrt{x^2}}\\
    &= \frac{x}{\abs{x}} = 
    \begin{cases} 
       \hfil \frac{x}{x} &\mbox{if } x>0 \\
        \frac{x}{-x} &\mbox{if } x<0
    \end{cases}\\
    &= \begin{cases} +1 &\mbox{if }x>0 \\ -1 &\mbox{if }x<0 \end{cases}\\
    \end{aligned}\\
\result{\abs{x}' = \sgnfn{x} = \frac{x}{\abs{x}}  \mbox{ if } x \neq 0} \label{eq:res:derivative_abs}
\end{gather}

\subsection{Reciprocal of $\sgnfn{x}$}
From definition of the $\sgn$ function (see: \ref{eq:sgn_def}) follows:
\begin{gather}
    \frac{1}{\sgnfn{x}} = {
        \begin{cases} 
            +1 &\mbox{if }x>0 \\
            \hfil 0 &\mbox{if }x=0 \\ 
            -1 &\mbox{if }x<0 
    \end{cases}}\\
\result{\frac{1}{\sgnfn{x}} = \sgnfn{x}} \label{eq:res:reciprocal_sgn}
\end{gather}
Using equation \ref{eq:res:derivative_abs}
\begin{gather}
\result{\sgnfn{x} = \frac{x}{\abs{x}} = \frac{\abs{x}}{x} \quad \mbox{if }x \neq 0}
\label{eq:sgn_abs_and_x}
\end{gather}

\subsection{$x$ and $\sgnfn{x}$}
From equation \ref{eq:sgn_abs_and_x} we get:
%
\begin{align}
    \resultnoimplies{\abs{x}=\sgnfn{x} \cdot x}\\
    \resultnoimplies{x=\sgnfn{x} \cdot \abs{x}}
\end{align}
when $x\neq 0$.

\subsection{Looking at $\sqrt{\dfrac{a}{x^2}}$}
\begin{gather}
\sqrt{\frac{a}{x^2}} = \sqrt{\frac{1}{x^2}} \cdot \sqrt{a} = \frac{1}{\sqrt{x^2}} \sqrt{a}\\
\result{\sqrt{\frac{a}{x^2}} = \frac{1}{\abs{x}}\sqrt{a}}
\label{eq:res:root_a_xsq}
\end{gather}

\subsection{Fun connnection with $x \cdot \sqrt{\dfrac{a}{x^2}}$}
From equation \ref{eq:res:root_a_xsq} we get:
\begin{gather}
    x \cdot \sqrt{\frac{a}{x^2}} = x \cdot \frac{1}{\abs{x}} \cdot \sqrt{a}
    = \frac{x}{\abs{x}} \cdot \sqrt{a}\\
    \result{x \cdot \sqrt{\frac{a}{x^2}} = \sgnfn{x} \cdot \sqrt{a}}
\end{gather}
