\section{Logarithms}
\label{Logarithmic_formulars}

\subsection{Recipricol of the Logarithm}
\label{1_over_log}

\begin{gather} \label{1_over_log_eqn}
\cfrac{1}{\log_a(b)}=\cfrac{1}{\cfrac{\log_c(b)}{\log_c(a)}}=\cfrac{\log_c(a)}{\log_c(b)}=\log_b(a) \\
\\
\result{\cfrac{1}{\log_a(b)} =\log_b(a)} \label{1_over_log_result}
\end{gather}



\subsection{Divison of logs}
\label{Division_of_logs}

\subsubsection{General case}
\label{log_over_log_general}
Using equation \ref{1_over_log_result} we get:

\begin{gather}
\resultnoimplies{\cfrac{ \log_a(b) } { \log_c(d) } = \log_a(b) \cdot \log_d(c) } \label{log_over_log_result}
\end{gather}

\subsubsection{same Argument}
\label{log_over_log_same_arg}

\noindent Using Equation \ref{log_over_log_result} and assuming that $b=d$, we get:
\begin{gather} \label{log_over_log_same_arg_eqn}
\cfrac{\log_a(b)} {\log_c(d)} = \cfrac{\log_a(b)} {\log_c(b)} 
= \log_a(b) \cdot \log_b(c) = \log_a( b^{ \log_b(c) } ) = \log_a(c) \\
\result{\cfrac{\log_a(b)} {\log_c(b)} = \log_a(c) }
\end{gather}
